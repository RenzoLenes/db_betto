// CREAR TABLA DE CATEGORIAS
db.createCollection(
    "categorias",
    { validator:
        { $jsonSchema:
                {   bsonType: "object",
                    description: "Document describing a category of dress",
                    required: ["nombre", "descripcion"],
                    properties:
                        {   "nombre":{
                            bsonType: "string",
                            description: "Name of the category"
                            },
                            "descripcion": {
                                bsonType: "string",
                                description: "description of the category"
                            },


                         }
                }
        }
    }
);
//INSERTAR DATO DE CATEGORIAS
db.categorias.insertOne(
{
nombre:"camisa",
descripcion:"prenda superior"
}
)
